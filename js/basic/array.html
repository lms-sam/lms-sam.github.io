# 随机排序

  1. tpl
function sort(arr){
    const l = arr.length;
    for(let i = 0; i < l; i++){
        const rc = parseInt(Math.random()*l);
        const tpl = arr[i];
        arr[i] = arr[rc];
        arr[rc] = tpl;
    }
    return arr;
}
  1. 随机索引抽取，插入新数组
function sort2(arr){
    var mixedArr = [];
    while (arr.length > 0) {
        let rc = parseInt(Math.random() * arr.length);
        mixedArr.push(arr.splice(rc));
        arr.splice(rc,1);
    }
    return mixedArr;
}
  1. sort
function sort3(arr){
    arr.sort(function(a,b){
        return Math.random()-0.5
    })
    return arr;
}

# 数组对象排序

  1. sort
function compareByProtoName(protoName) {
    return function(a,b) {
        return a[protoName] - b[protoName]
    }
}
function sort(ObjectArr,protoName) {
    ObjectArr.sort(compareByProtoName(protoName));
    return ObjectArr;
}
