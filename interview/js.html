# 一、 隐式转换

# 介绍

# 规则

1.1 字符串连接 1.2 转成number类型 1.3 转成boolean类型

# 常见问题

# 1. 字符串连接符与算术运算符隐式转换规则混淆 （+号）

// 字符串连接符：存在字符串+号连接左右。
console.log(1 + 'true'); // '1true'
console.log('true' + 1); // 'true1'
// 隐式转换
console.log(1 + true); // 2
console.log(1 + undefined); // NaN;
console.log(1 + null); // 1

# 2. 关系运算符：会把其他数据类型转换成number之后再比较关系 （>、<、==）

// 数字跟字符串对比，先转换字符串为number
console.log('2' > 10); // false
// 都是字符串对比unicode编码顺序，由相同位数比较后，相同就下一位比较
console.log('2' > '10'); // true
console.log('abc' > 'b'); // false
console.log('aab' > 'ab'); // false
// 无视规则，NaN不等于任何值
console.log(NaN == NaN); // false
// 无视任何规则 undefined = undefined = null = null;
console.log(undefined == null); // true

# 3. 复杂数据类型在隐式转换时会先转成String，然后再转成Number运算 (valueOf)

var a = '?';
if (a == 1 && a == 2 & a == 3) {
    console.log('in');
};
// 答: 解析对象/数组，先会调用valueOf,然后再调用toString,复写valueOf可以实现以上问题
a = {
    i: 0;
    valueOf: function () {
        return ++a.i
    }
}
// eg
console.log([1, 2] == '1, 2');
console.log({} == '[object Object]');

# 4. 逻辑非隐式转换与关系运算符隐式转换搞混淆 (!先将数据用Boolean转换)

console.log( ![] == false); // true (Boolean([]) = true)
console.log( [] == false); // true [].valueOf().toString() = ''

console.log( !{} == {}); // true (Boolean({}) = true)
console.log( {} == {}); // 相同类型不转换，当前只存堆地址。
