# this的指向

绑定的方式

  1. 隐性绑定 xx.fn();
  2. 显性绑定 apply/call
    1. 硬绑定 bind
  3. new 绑定 构造函数
  4. 默认绑定 当前的上下文

优先级

硬绑定 > 显示绑定 > 隐式绑定 > new绑定 > 默认绑定

function Foo() {
    this.name = 'className';
}
var obj = {
    name: 'objName',
    type: 'obj1'
}
function sayHi(name) {
    this.name = name;
    console.log(this);
}
Foo.prototype.sayHi = sayHi.bind(obj);
Foo.prototype.sayHello = sayHi;
var fo = new Foo();
fo.sayHi('aaa');

var obj2 = {
    name: '',
    type: '2',
    fn: fo.sayHello
}
obj2.fn('eee');

# 1. 隐性绑定 -> 当前上级的调用对象

var name = 'sam';
function foo (name) {
    this.name = name;
}
foo('小强'); // window.name = '小强';
console.log(name);
var obj = {
    name: 'lili',
    fn: foo
}
obj.fn('小雯');// obj.name = '小雯';
console.log(obj);

# 2. 显性绑定 -> 指定对象，null或者undefined变成默认绑定

var name = 'sam';
let obj = {
    name: 'George'
}
function foo (name) {
    this.name = name
}

foo('eric'); // window.name = 'eric';
console.log(name);
foo.call(obj, '小丽'); // obj.name = '小丽'；
console.log(obj);
foo.apply(obj, ['小强']); // obj.name = '小强';
console.log(obj);

var bar = foo.bind(obj);

bar('小雯'); // obj.name = '小雯';
console.log(obj);

# 3. new绑定->实例

function Foo() {
    this.name = 'className';
}
Foo.prototype.sayHi = function sayHi(name) {
    this.name = name;
    console.log(this); // this是绑定的实例
}

# 4. 默认绑定

当前作用域

function foo () {
    console.log(this); // window;
}

var obj = {
    fn: function() {
        console.log(this); // obj
        function bar () {
            console.log(this); // window;
        }
        bar();
    }
}
obj.fn();
